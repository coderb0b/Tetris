1   package my.game.tetrisproject.gui;
2   
3   import java.awt.event.KeyEvent;
4   import java.awt.event.KeyListener;
5   import my.game.tetrisproject.domain.Tetromino;
6   import my.game.tetrisproject.logic.Game;
7   
8   /**
9    * Luokka kuuntelee näppäimistön komentoja ja käskyttää näiden perusteella
10   * peliä ja tetrominoa.
11   */
12  public class KeyboardListener implements KeyListener {
13  
14      private final Game game;
15  
16      public KeyboardListener(Game game) {
17          this.game = game;
18      }
19  
20      @Override
21      public void keyTyped(KeyEvent e) {
22  
23      }
24  
25      /**
26       * Kuuntelee painettuja näppäinkomentoja.
27       * @param e 
28       */
29      @Override
30      public void keyPressed(KeyEvent e) {
31          Tetromino active = game.getBoard().getCurrentTetro();
32          
33          
34          if (e.getKeyCode() == KeyEvent.VK_UP) {            
35              active.rotateLeft();
36          } else if (e.getKeyCode() == KeyEvent.VK_DOWN) {
37              active.rotateRight();
38          } else if (e.getKeyCode() == KeyEvent.VK_RIGHT) {
39              this.game.tryMove(active.getTetroX() + 1, active.getTetroY());
40          } else if (e.getKeyCode() == KeyEvent.VK_LEFT) {
41              this.game.tryMove(active.getTetroX() - 1, active.getTetroY());
42          } else if (e.getKeyCode() == KeyEvent.VK_SPACE) {
43              this.game.moveDown();
44          } else if (e.getKeyCode() == KeyEvent.VK_D) {
45              this.game.moveOneDown();
46          }
47      }
48  
49      @Override
50      public void keyReleased(KeyEvent e) {
51      }
52  
53  }
