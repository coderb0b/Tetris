1   package my.game.tetrisproject.gui;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import javax.swing.JPanel;
6   import my.game.tetrisproject.domain.Block;
7   import my.game.tetrisproject.logic.Board;
8   import my.game.tetrisproject.logic.Game;
9   
10  /**
11   *Piirtoalustaluokka vastaa piirrettävistä muodoista.
12   */
13  public class TRenderer extends JPanel implements Updater {
14  
15      private Board board;
16      private Game game;
17      private int scale = 20;
18  
19      public TRenderer(Game peli) {
20          this.board = peli.getBoard();
21          this.game = peli;
22  
23      }
24  
25      public Game getGame() {
26          return this.game;
27      }
28  
29      @Override
30      protected void paintComponent(Graphics g) {
31          super.paintComponent(g);
32  
33          //Pelilaudan kehys
34          g.setColor(Color.BLACK);
35          g.drawRect(0, 0, board.getWidth() * this.scale, board.getHeight() * this.scale);
36  
37          drawBoard(g);
38      }
39  
40      private void drawBoard(Graphics g) {
41          for (int i = 0; i < board.getWidth(); i++) {
42              for (int j = 0; j < board.getHeight(); j++) {
43                  //piirretään aktiiviset Tetrominot tällä.. Kesken..
44                  char shape = this.board.getShapeFromBoard(i, j);
45                  Block b = this.board.getStationaryBlock(i, j);
46                  if (shape == 'F') {
47                      g.setColor(board.getCurrentTetro().getBlocks().get(0).getColor());
48                      g.fillRect(i * scale + 1, j * scale + 1, scale, scale);
49  
50                  } else if (b != null) {
51                      g.setColor(b.getColor());
52                      g.fillRect(i * scale + 1, j * scale + 1, scale, scale);
53                  }
54              }
55          }
56  
57          //piirretään paikallaan olevat palikat
58          if (board.getCurrentTetro().getShape() != 'X') {
59              for (Block b : board.getCurrentTetro().getBlocks()) {
60                  int x = board.getCurrentTetro().getTetroX() + b.getX();
61                  int y = board.getCurrentTetro().getTetroY() - b.getY();
62                  g.setColor(b.getColor());
63                  g.fillRect(x * scale + 1, y * scale + 1, scale, scale);
64              }
65          }
66      }
67  
68      /**
69       * Päivitetään näkymää, kutsutaan peliluupissa
70       */
71      @Override
72      public void update() {
73          this.repaint();
74      }
75  
76  }
